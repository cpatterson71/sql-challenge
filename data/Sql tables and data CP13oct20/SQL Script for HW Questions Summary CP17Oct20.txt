select * from employees;

create table salaries(
	emp_no INT,
	salary INT);
	
select e.emp_no, e.last_name, e.first_name, e.sex, s.salary from employees As e
left join salaries As s on (e.emp_no = s.emp_no)
order by e.emp_no;
